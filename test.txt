from airflow.providers.microsoft.mssql.hooks.mssql import MsSqlHook
from airflow.exceptions import AirflowException

def verificar_e_inserir_dados_sql_server():
    # Configurar o ID da conexão criada no Airflow
    mssql_conn_id = 'minha_conexao_sql_server'

    # Dados para verificar e inserir
    tabela = "minha_tabela"
    chave_unica = "coluna_unica"  # A coluna que deve ser única
    valores = [
        {"coluna_unica": "valor1", "coluna2": "valor2"},
        {"coluna_unica": "valor3", "coluna2": "valor4"}
    ]

    try:
        # Criar o hook para o SQL Server
        mssql_hook = MsSqlHook(mssql_conn_id=mssql_conn_id)
        
        for registro in valores:
            valor_chave = registro[chave_unica]

            # Verificar se o valor já existe
            query_verificacao = f"SELECT COUNT(*) FROM {tabela} WHERE {chave_unica} = %s"
            resultado = mssql_hook.get_first(query_verificacao, parameters=(valor_chave,))
            
            if resultado[0] > 0:
                print(f"O valor '{valor_chave}' já existe na tabela '{tabela}'. Pulando inserção.")
                continue

            # Inserir o registro se não existir
            colunas = ", ".join(registro.keys())
            placeholders = ", ".join([f"%({coluna})s" for coluna in registro.keys()])
            query_insercao = f"INSERT INTO {tabela} ({colunas}) VALUES ({placeholders})"
            mssql_hook.run(sql=query_insercao, parameters=registro)
            print(f"Registro com '{valor_chave}' inserido com sucesso!")

    except AirflowException as e:
        print(f"Erro no Airflow ao verificar/inserir dados: {e}")
    
    except Exception as e:
        print(f"Ocorreu um erro inesperado: {e}")
